474  private void leerMensajeXBee(XBeeMessage mensajeXBee) {
475  byte[] arrayBytes = mensajeXBee.getData();
476  int dato = arrayBytes[0];
477  double velocidad;
478 
479  switch (dato) {
480  case DispositivoXBee.VUELTAS_RUEDA_PRACTICA:
481  DecimalFormat formatoDecimal = new DecimalFormat("0.00");
482 
483  distanciaRecorrida += Math.PI * coche.getDiametroRueda() * 0.01;
484  textoDistanciaRecorrida.setText(formatoDecimal.format(distanciaRecorrida) + " m");
485 
486  velocidad = Fisica.calcularVelocidadMedia(distanciaRecorrida, ((cronometro.getMinuto() * 60)
487  + cronometro.getSegundo() + ((cronometro.getMilesima() - 1) / 1000)));
488  textoVelocidad.setText(formatoDecimal.format(velocidad) + " Km/h");
489 
490  vueltasRueda++;
491  textoVueltasRueda.setText(String.valueOf(vueltasRueda));
492 
493  establecerPorcentaje();
494 
495  establecerServo();
496  break;
497 
498  case DispositivoXBee.OBSTACULO_PRACTICA:
499  obstaculo = true;
500  textoObstaculo.setText("SÍ");
501  detenerPractica();
502  break;
503 
504  case DispositivoXBee.FINALIZAR_PRACTICA:
505  detenerPractica();
506  break;
507 
508  default:
509  dispositivoXBee.enviarDatosCircuito(DispositivoXBee.MOTOR, DispositivoXBee.DETENER, DispositivoXBee.DETENER,
510  DispositivoXBee.DETENER);
511 
512  detenerPractica();
513  new DialogoOpcionesAlerta(ventana, "¡Datos incorrectos!", ERROR);
514  break;
515  }
516  }
